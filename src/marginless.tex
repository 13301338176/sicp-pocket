% Use @pocket for e-reader formatting (based on smallbook).
% Contributed by Gavrie Philipson. Modified by Andres Raba.
\def\pocket{{\globaldefs = 1
  \parskip = 2pt plus 1pt
  \textleading = 21.3pt		% 33%
  %
  % Parameters in order: 1) textheight; 2) textwidth;
  % 3) voffset; 4) hoffset; 5) binding offset; 6) topskip;
  % 7) physical page height; 8) physical page width.

  \internalpagesizes{7.92in}{5.8in}%
                    {-0.87in}{-0.81in}%
                    {0pt}{0pt}% 
                    {8.24in}{6.18in}%
  
  \lispnarrowing = 0.3in
  \tolerance = 700
  \hfuzz = 1pt
  \contentsrightmargin = 0pt
  \defbodyindent = .5cm
}}

% Restore the Texinfo character set.
\catcode`\\=\active
